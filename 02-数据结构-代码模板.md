## 单链表 —— AcWing 826

```cpp
int head, e[N], ne[N], idx;

void init()
{
    head = -1;
    idx = 0;
}

void insert(int a)  // 头插
{
    e[idx] = a, ne[idx] = head, head = idx ++ ;
}

void remove()  // 删头结点
{
    head = ne[head];
}
```

## 双链表 —— AcWing 827

```cpp
int e[N], l[N], r[N], idx;

void init()
{
    r[0] = 1, l[1] = 0;
    idx = 2;
}

void insert(int a, int x)  // 在a右边插入x
{
    e[idx] = x;
    l[idx] = a, r[idx] = r[a];
    l[r[a]] = idx, r[a] = idx ++ ;
}

void remove(int a)
{
    l[r[a]] = l[a];
    r[l[a]] = r[a];
}
```

## Trie树 —— AcWing 835

```cpp
int son[N][26], cnt[N], idx;

void insert(char *str)
{
    int p = 0;
    for (int i = 0; str[i]; i ++ )
    {
        int u = str[i] - 'a';
        if (!son[p][u]) son[p][u] = ++ idx;
        p = son[p][u];
    }
    cnt[p] ++ ;
}

int query(char *str)
{
    int p = 0;
    for (int i = 0; str[i]; i ++ )
    {
        int u = str[i] - 'a';
        if (!son[p][u]) return 0;
        p = son[p][u];
    }
    return cnt[p];
}
```

## 并查集 —— AcWing 836, 837

```cpp
// (1) 朴素并查集
int p[N];
int find(int x)
{
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}
// 初始化：for (int i = 1; i <= n; i++) p[i] = i;
// 合并：p[find(a)] = find(b);

// (2) 维护size
int p[N], size[N];
int find(int x)
{
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}
// 合并：size[find(b)] += size[find(a)]; p[find(a)] = find(b);

// (3) 维护到祖宗节点距离
int p[N], d[N];
int find(int x)
{
    if (p[x] != x)
    {
        int u = find(p[x]);
        d[x] += d[p[x]];
        p[x] = u;
    }
    return p[x];
}
```

## 堆 —— AcWing 838, 839

```cpp
int h[N], ph[N], hp[N], size;

void heap_swap(int a, int b)
{
    swap(ph[hp[a]], ph[hp[b]]);
    swap(hp[a], hp[b]);
    swap(h[a], h[b]);
}

void down(int u)
{
    int t = u;
    if (u * 2 <= size && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= size && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t)
    {
        heap_swap(u, t);
        down(t);
    }
}

void up(int u)
{
    while (u / 2 && h[u] < h[u / 2])
    {
        heap_swap(u, u / 2);
        u >>= 1;
    }
}

// O(n)建堆
for (int i = n / 2; i; i -- ) down(i);
```

## 哈希表 —— AcWing 840

```cpp
// (1) 拉链法
int h[N], e[N], ne[N], idx;

void insert(int x)
{
    int k = (x % N + N) % N;
    e[idx] = x;
    ne[idx] = h[k];
    h[k] = idx ++ ;
}

bool find(int x)
{
    int k = (x % N + N) % N;
    for (int i = h[k]; i != -1; i = ne[i])
        if (e[i] == x) return true;
    return false;
}

// (2) 开放寻址法
int h[N];
int find(int x)
{
    int t = (x % N + N) % N;
    while (h[t] != null && h[t] != x)
    {
        t ++ ;
        if (t == N) t = 0;
    }
    return t;
}
```

## 字符串哈希 —— AcWing 841

```cpp
// P取131或13331，用unsigned long long自动取模2^64
typedef unsigned long long ULL;
ULL h[N], p[N];

// 初始化
p[0] = 1;
for (int i = 1; i <= n; i ++ )
{
    h[i] = h[i - 1] * P + str[i];
    p[i] = p[i - 1] * P;
}

// 求子串str[l~r]的哈希值
ULL get(int l, int r)
{
    return h[r] - h[l - 1] * p[r - l + 1];
}
```
